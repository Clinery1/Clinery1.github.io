We can also do other languages like GLSL:

#version 450


layout(location=0) in vec2 v_uv;
layout(location=1) in vec4 v_color;

layout(location=0) out vec4 f_color;

layout(set=0,binding=0) uniform sampler2D tex;


void main() {
    vec4 tex_color=texture(tex,v_uv);
    f_color=vec4(v_color.rgb,v_color.a*tex_color.r);
}

or assembly:

global _start


section .data
msg:db "Hello, World!",0xa
msgLen: equ $-msg


section .text
_start:
    mov rax,1
    mov rdi,1
    mov rsi,msg
    mov rdx,msgLen
    syscall
    mov rax,60
    mov rdi,0
    syscall
